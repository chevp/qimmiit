VKAPI_ATTR VkBool32 VKAPI_CALL debugMessageCallback(VkDebugReportFlagsEXT flags, VkDebugReportObjectTypeEXT objType, uint64_t srcObject, size_t location, int32_t msgCode, const char *pLayerPrefix, const char *pMsg, void *pUserData)
{
	std::string prefix("");
	if (flags & VK_DEBUG_REPORT_ERROR_BIT_EXT)
	{
		prefix += "ERROR:";
	};
	if (flags & VK_DEBUG_REPORT_WARNING_BIT_EXT)
	{
		prefix += "WARNING:";
	};
	if (flags & VK_DEBUG_REPORT_DEBUG_BIT_EXT)
	{
		prefix += "DEBUG:";
	}

	std::string debugMessage = prefix + " [" + std::string(pLayerPrefix) + "] Code " +
							   std::to_string(msgCode) + " : " + std::string(pMsg);

	// Enhanced logging with ocean logging system
	ocean::info("=== VULKAN DEBUG CALLBACK ===");
	ocean::info("Object Type: " + std::to_string(objType) + ", Source Object: " + std::to_string(srcObject));
	ocean::info("Location: " + std::to_string(location) + ", Message Code: " + std::to_string(msgCode));

	if (flags & VK_DEBUG_REPORT_ERROR_BIT_EXT) {
		ocean::error("VULKAN VALIDATION ERROR: " + debugMessage);
	} else if (flags & VK_DEBUG_REPORT_WARNING_BIT_EXT) {
		ocean::warn("VULKAN VALIDATION WARNING: " + debugMessage);
	} else {
		ocean::info("VULKAN VALIDATION DEBUG: " + debugMessage);
	}

#if defined(__ANDROID__)
	LOGD("%s", debugMessage.c_str());
#else
	std::cout << debugMessage << "\n";
#endif
	fflush(stdout);
	return VK_FALSE;
}

	VkResult VulkanExampleBase::createInstance(bool enableValidation)
	{
		ocean::info("=== VULKAN INSTANCE CREATION START ===");
		ocean::info("Application name: " + name);
		ocean::info("Validation requested: " + std::string(enableValidation ? "true" : "false"));

		this->settings.validation = enableValidation;

		// Validation can also be forced via a define
#if defined(_VALIDATION)
		this->settings.validation = true;
		ocean::info("Validation forced via _VALIDATION define");
#endif

		ocean::info("Final validation setting: " + std::string(this->settings.validation ? "enabled" : "disabled"));

		VkApplicationInfo appInfo = {};
		appInfo.sType = VK_STRUCTURE_TYPE_APPLICATION_INFO;
		appInfo.pApplicationName = name.c_str();
		appInfo.pEngineName = name.c_str();
		appInfo.apiVersion = VK_API_VERSION_1_0;

		ocean::info("Vulkan API version: " + std::to_string(VK_VERSION_MAJOR(VK_API_VERSION_1_0)) + "." +
				   std::to_string(VK_VERSION_MINOR(VK_API_VERSION_1_0)) + "." +
				   std::to_string(VK_VERSION_PATCH(VK_API_VERSION_1_0)));

		std::vector<const char *> instanceExtensions = {VK_KHR_SURFACE_EXTENSION_NAME};
		ocean::info("Base extensions: VK_KHR_SURFACE_EXTENSION_NAME");

		// Refactored: Extract platform-specific extension logic
		addPlatformSpecificExtensions(instanceExtensions);

		if (settings.validation)
		{
			// Refactored: Extract validation setup logic
			addValidationExtensions(instanceExtensions);
		}

		// Create Vulkan instance
		VkInstanceCreateInfo instanceCreateInfo = {};
		instanceCreateInfo.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
		instanceCreateInfo.pNext = NULL;
		instanceCreateInfo.pApplicationInfo = &appInfo;

		if (instanceExtensions.size() > 0)
		{
			instanceCreateInfo.enabledExtensionCount = static_cast<uint32_t>(instanceExtensions.size());
			instanceCreateInfo.ppEnabledExtensionNames = instanceExtensions.data();
		}

		std::vector<const char *> validationLayers;
		if (settings.validation)
		{
			// Refactored: Extract validation layer setup
			setupValidationLayers(validationLayers, instanceCreateInfo);
		}

		ocean::info("Creating Vulkan instance...");
		VkResult result = vkCreateInstance(&instanceCreateInfo, nullptr, &instance);

		if (result != VK_SUCCESS) {
			ocean::error("Failed to create Vulkan instance! Result: " + std::to_string(result));
			return result;
		}

		ocean::info("Vulkan instance created successfully");

		// Setup debug callback after instance creation
		if (settings.validation) {
			setupDebugCallback();
		}

		ocean::info("=== VULKAN INSTANCE CREATION COMPLETE ===");
		return result;
	}

	// Refactored method: Extract platform-specific extension logic
	void VulkanExampleBase::addPlatformSpecificExtensions(std::vector<const char*>& instanceExtensions)
	{
#if defined(_WIN32)
		instanceExtensions.push_back(VK_KHR_WIN32_SURFACE_EXTENSION_NAME);
		ocean::info("Platform: Windows, adding VK_KHR_WIN32_SURFACE_EXTENSION_NAME");
#elif defined(VK_USE_PLATFORM_ANDROID_KHR)
		instanceExtensions.push_back(VK_KHR_ANDROID_SURFACE_EXTENSION_NAME);
		ocean::info("Platform: Android, adding VK_KHR_ANDROID_SURFACE_EXTENSION_NAME");
#elif defined(_DIRECT2DISPLAY)
		ocean::info("Platform: Direct2Display (no additional surface extensions needed)");
#elif defined(VK_USE_PLATFORM_WAYLAND_KHR)
		instanceExtensions.push_back(VK_KHR_WAYLAND_SURFACE_EXTENSION_NAME);
		ocean::info("Platform: Wayland, adding VK_KHR_WAYLAND_SURFACE_EXTENSION_NAME");
#elif defined(VK_USE_PLATFORM_XCB_KHR)
		instanceExtensions.push_back(VK_KHR_XCB_SURFACE_EXTENSION_NAME);
		ocean::info("Platform: XCB, adding VK_KHR_XCB_SURFACE_EXTENSION_NAME");
#elif defined(VK_USE_PLATFORM_MACOS_MVK)
		instanceExtensions.push_back(VK_MVK_MACOS_SURFACE_EXTENSION_NAME);
		ocean::info("Platform: macOS, adding VK_MVK_MACOS_SURFACE_EXTENSION_NAME");
#endif
	}

	// Refactored method: Extract validation extension logic
	void VulkanExampleBase::addValidationExtensions(std::vector<const char*>& instanceExtensions)
	{
		instanceExtensions.push_back(VK_EXT_DEBUG_REPORT_EXTENSION_NAME);
		ocean::info("Validation: Adding VK_EXT_DEBUG_REPORT_EXTENSION_NAME");
	}

	// Refactored method: Extract validation layer setup
	void VulkanExampleBase::setupValidationLayers(std::vector<const char*>& validationLayers, VkInstanceCreateInfo& instanceCreateInfo)
	{
		validationLayers.push_back("VK_LAYER_KHRONOS_validation");
		instanceCreateInfo.enabledLayerCount = static_cast<uint32_t>(validationLayers.size());
		instanceCreateInfo.ppEnabledLayerNames = validationLayers.data();
		ocean::info("Validation: Adding VK_LAYER_KHRONOS_validation");
	}

	// Refactored method: Extract debug callback setup
	void VulkanExampleBase::setupDebugCallback()
	{
		vkCreateDebugReportCallback = reinterpret_cast<PFN_vkCreateDebugReportCallbackEXT>(vkGetInstanceProcAddr(instance, "vkCreateDebugReportCallbackEXT"));
		vkDestroyDebugReportCallback = reinterpret_cast<PFN_vkDestroyDebugReportCallbackEXT>(vkGetInstanceProcAddr(instance, "vkDestroyDebugReportCallbackEXT"));

		VkDebugReportCallbackCreateInfoEXT dbgCreateInfo = {};
		dbgCreateInfo.sType = VK_STRUCTURE_TYPE_DEBUG_REPORT_CREATE_INFO_EXT;
		dbgCreateInfo.pfnCallback = (PFN_vkDebugReportCallbackEXT)debugMessageCallback;
		dbgCreateInfo.flags = VK_DEBUG_REPORT_ERROR_BIT_EXT | VK_DEBUG_REPORT_WARNING_BIT_EXT | VK_DEBUG_REPORT_PERFORMANCE_WARNING_BIT_EXT;

		VkResult err = vkCreateDebugReportCallback(instance, &dbgCreateInfo, nullptr, &debugReportCallback);
		if (err == VK_SUCCESS) {
			ocean::info("Debug callback created successfully");
		} else {
			ocean::error("Failed to create debug callback! Result: " + std::to_string(err));
		}
	}

	VulkanExampleBase::VulkanExampleBase()
	{
		char *numConvPtr;
		// Parse command line arguments
		for (size_t i = 0; i < args.size(); i++)
		{
			if (args[i] == std::string("-validation"))
			{
				settings.validation = true;
			}
			if (args[i] == std::string("-vsync"))
			{
				settings.vsync = true;
			}
			if ((args[i] == std::string("-f")) || (args[i] == std::string("--fullscreen")))
			{
				settings.fullscreen = true;
			}
			if ((args[i] == std::string("-w")) || (args[i] == std::string("--width")))
			{
				uint32_t w = strtol(args[i + 1], &numConvPtr, 10);
				if (numConvPtr != args[i + 1])
				{
					width = w;
				};
			}
			if ((args[i] == std::string("-h")) || (args[i] == std::string("--height")))
			{
				uint32_t h = strtol(args[i + 1], &numConvPtr, 10);
				if (numConvPtr != args[i + 1])
				{
					height = h;
				};
			}
		}

#if defined(VK_USE_PLATFORM_ANDROID_KHR)
		// Vulkan library is loaded dynamically on Android
		bool libLoaded = vks::android::loadVulkanLibrary();
		assert(libLoaded);
#elif defined(_DIRECT2DISPLAY)

#elif defined(VK_USE_PLATFORM_WAYLAND_KHR)
		initWaylandConnection();
#elif defined(VK_USE_PLATFORM_XCB_KHR)
		initxcbConnection();
#endif

#if defined(_WIN32)
		AllocConsole();
		AttachConsole(GetCurrentProcessId());
		FILE *stream;
		freopen_s(&stream, "CONOUT$", "w+", stdout);
		freopen_s(&stream, "CONOUT$", "w+", stderr);
		SetConsoleTitle(TEXT("Vulkan validation output"));
#endif

		// Initialize utility classes
		inputHandler = std::make_unique<coregfx::platform::InputHandler>();
		framebufferManager = std::make_unique<coregfx::platform::FramebufferManager>();
	}
